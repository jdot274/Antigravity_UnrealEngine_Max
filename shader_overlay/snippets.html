struct Spinlock {
    std::atomic_flag flag = ATOMIC_FLAG_INIT;
    void lock() {
        while (flag.test_and_set(std::memory_order_acquire));
    }
    void unlock() {
        flag.clear(std::memory_order_release);
    }
};

__m128 intersect(Ray r, AABB box) {
    __m128 invDir = _mm_rcp_ps(r.dir);
    __m128 t1 = _mm_mul_ps(_mm_sub_ps(box.min, r.ori), invDir);
    __m128 t2 = _mm_mul_ps(_mm_sub_ps(box.max, r.ori), invDir);
    return _mm_min_ps(_mm_max_ps(t1, t2), ...);
}
